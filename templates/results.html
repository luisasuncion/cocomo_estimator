Detalle del calculo

Esfuerzo

PM = 2.94 × (KSLOC)^E × EAF

PM = {{ "%.2f"|format(result.constant_a) }} × ({{ result.ksloc }})^{{ "%.4f"|format(result.exponent_e) }} × {{ "%.4f"|format(result.eaf) }}

PM = {{ "%.2f"|format(result.effort_pm) }} personas-mes

Exponente de tiempo

F = 0.28 + 0.2 × (E - 0.91)

F = {{ "%.2f"|format(result.constant_d) }} + 0.2 × ({{ "%.4f"|format(result.exponent_e) }} - {{ "%.2f"|format(result.constant_b) }})

F = {{ "%.4f"|format(result.schedule_exponent_f) }}

Tiempo nominal

TDEV = 3.67 × PM^F

TDEV = {{ "%.2f"|format(result.constant_c) }} × ({{ "%.4f"|format(result.effort_pm) }})^{{ "%.4f"|format(result.schedule_exponent_f) }}

TDEV = {{ "%.2f"|format(result.tdev_months) }} meses

Personal promedio

Personal = PM / TDEV

Personal = {{ "%.4f"|format(result.effort_pm) }} / {{ "%.2f"|format(result.tdev_months) }}

Personal = {{ "%.2f"|format(result.average_staff) }} personas

Costo total

Costo = PM × salario mensual promedio

Costo = {{ "%.4f"|format(result.effort_pm) }} × {{ "%.2f"|format(result.average_monthly_salary) }}

Costo = S/ {{ result.total_cost|currency_amount }}
